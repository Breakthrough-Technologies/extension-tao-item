 11 /** 
 12  * The QTIResultCollector class collects the user results of a QTI widgets defineby the options
 13  * @namespace QTI
 14  * @class QTIResultCollector
 15  * @param {Object} options
 16  */
 17 function QTIResultCollector(options){
 18 
 19 	var _this = this;
 20 	
 21 	this.opts = options;
 22 	this.id = options['id'];
 23 	
 24 	// result process
 25 	this.choice = function(){
 26 		var result = {
 27 			"identifier"	: _this.opts['responseIdentifier'] // Identifier of the response
 28 			, "cardinality" : _this.opts["maxChoices"] << 1 != 2 ? 'multiple' : 'single'
 29 			, "type"		: _this.id
 30 			, "values"		: []
 31 		};
 32 		
 33 		var userData = new Array();
 34 		$("#" + _this.id + " .tabActive").each(function(){
 35 			userData.push(this.id);
 36 		});
 37 		result.values = userData;
 38 		
 39 		return result;
 40 	};
 41 	
 42 	this.order = function (){
 43 		var result = new Array();
 44 		$("#" + _this.id + " ul.qti_choice_list li").each(function(){
 45 			result.push(this.id);
 46 		});
 47 		return result;
 48 	};
 49 
 50 	this.associate = function(){
 51 		var result = new Array();
 52 		
 53 		$("#" + _this.id + " .qti_association_pair").each(function(){
 54 			result.push([$(this).find('li:first').attr('id'), $(this).find('li:last').attr('id')]);
 55 		});
 56 		return result;
 57 	};
 58 
 59 	this.text = function(){
 60 		
 61 		//single mode
 62 		if($("#" + _this.id ).get(0).nodeName.toLowerCase() != 'div'){
 63 			return new Array($("#" + id).val());
 64 		}
 65 		
 66 		//multiple mode
 67 		var result = new Array();
 68 		$("#" + _this.id + " :text").each(function(){
 69 			result.push($(this).val());
 70 		});
 71 		return result;
 72 	};
 73 	this.text_entry = this.text;
 74 	this.extended_text = this.text;
 75 
 76 	this.inline_choice = function(){
 77 		return [$("#" + _this.id).val()];
 78 	};
 79 
 80 	this.hottext = function(){
 81 		var result = new Array();
 82 		$("#" + _this.id + " .hottext_choice_on").each(function(){
 83 			result.push(this.id.replace("/^hottext_choice_/", ''));
 84 		});
 85 		return result;
 86 	};
 87 
 88 	this.gap_match = function(){
 89 		var result = new Array();
 90 		$("#" + _this.id + " .filled_gap").each(function(){
 91 			result.push([$(this).attr('id').replace('gap_', ''), $(this).parent().attr('id')]);
 92 		});
 93 		return result;
 94 	};
 95 	
 96 	this.match = function(){
 97 		var result = new Array();
 98 		$("#" + _this.id + " .tabActive").each(function(){
 99 			var subset = new Array();
100 			var classes = $(this).attr('class').split(' ');
101 			if(classes.length > 0){
102 				var i = 0;
103 				while(i < classes.length){
104 					if(/^xnode_/.test(classes[i])){
105 						subset[0] = classes[i].replace('xnode_', '');
106 					}
107 					if(/^ynode_/.test(classes[i])){
108 						subset[1] = classes[i].replace('ynode_', '');
109 					}
110 					i++;
111 				}
112 				result.push(subset);
113 			}
114 		});
115 		return result;
116 	};
117 }
