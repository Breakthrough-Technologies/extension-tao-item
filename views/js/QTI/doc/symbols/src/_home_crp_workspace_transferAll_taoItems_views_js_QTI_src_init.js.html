 13 /**
 14  * The qti_initParam var is used everywhere in the QTI document to collect the interactions parameters
 15  * @example <code>qti_initParam['interaction_serial_1234'] = {id : 'interaction_1', type : 'qti_order_interaction',  responseIdentifier : 'RESPONSE'}</code> 
 16  * @type {Object}
 17  */
 18 var qti_initParam  	= new Object();
 19 
 20 /**
 21  * Initialize the QTI environment
 22  * 
 23  * @param {Object} qti_initParam the parameters of ALL item's interaction
 24  * @return void
 25  */
 26 function qti_init(qti_initParam){
 27 	for (var a in qti_initParam){
 28 		qti_init_interaction(qti_initParam[a]);
 29 	}
 30 }
 31 
 32 /**
 33  * Initialize the widget and the result collection for an interaction
 34  * 
 35  * @param {Object} initObj the params of the interaction (parts of qti_initParam identified by the interaction id)
 36  */
 37 function qti_init_interaction(initObj){
 38 	
 39 	//instantiate the widget class with the given interaction parameters
 40 	var myQTIWidget = new QTIWidget(initObj);
 41 	
 42 	//instantiate the result class with the given interaction parameters
 43 	var myResultCollector = new QTIResultCollector(initObj);
 44 	
 45 	//get the interaction type to identify the method 
 46 	var typeName = initObj["type"].replace('qti_', '').replace('_interaction', '');
 47 	
 48 	/** @todo remove it in prod ! */
 49 	if(!myQTIWidget[typeName]){
 50 		alert("Error: Unknow widget " + typeName);
 51 	}
 52 	
 53 	//call the widget initialization method
 54 	myQTIWidget[typeName].apply();
 55 	
 56 	// validation process
 57 	$("#qti_validate").bind("click",function(e){
 58 		e.preventDefault();
 59 		
 60 		$('body').css('cursor', 'wait');
 61 		
 62 		// Get user's data
 63 		var result = myResultCollector[typeName].apply();
 64 		// Set the matching engine with the user's data	
 65 		if(typeof(matchingSetResponses) == 'function'){
 66 			matchingSetResponses ([result]);
 67 		}
 68 	});
 69 }
