 14 TAO_MATCHING.Ellipse.prototype = {

 16         var returnValue = false;

 18         var point_y = (point.getValue())[1].getValue();
 19         var center_x = (this.center.getValue())[0].getValue();
 20         var center_y = (this.center.getValue())[1].getValue();
 21         var a = Math.pow(point_x-center_x,2)/Math.pow(this.hradius,2);

 25     }
